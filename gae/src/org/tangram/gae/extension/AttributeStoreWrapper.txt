package org.tangram.gae.extension;

import java.util.Collection;

public class AttributeStoreWrapper {
    
    AttributeStore store;
    
    String customerId;
    
    public AttributeStoreWrapper(AttributeStore store, Object o) {
        // FIXME: won't work persistently
        customerId = ""+o.hashCode();
        this.store = store;
    } // AttributeStoreWrapper()
    
    public Collection<Object> getValues(String contentId, String key) {
        return store.getValues(contentId, customerId, key);
    } // getValues()
    
    public void setValues(String contentId, String key, Collection<Object> values) {
        store.setValues(contentId, customerId, key, values);
    } // setValues()
    
} // AttributeStoreWrapper()
